RUMUS EXCEL DASAR

MAX

=MAX(A1-A4)
mengembalikan nilai terbesar mulai dari cell A1 s/d A4

MIN

=MIN(A1-A4)
mengembalikan nilai terkecil mulai dari cell A1 s/d A4

IF

=IF(Kriteria, nilai-jika-true, nilai-jika-false)
=IF(A1="APEL", "BUAH-BUAHAN", "BUKAN BUAH-BUAHAN")
jika nilai A1 adalah Apel kembalikan nilai "BUAH - BUAHAN' jika tidak kembalikan nilia "BUKAN BUAH - BUAHAN"

SUM

=SUM(A1-A3)
jumlahkan angka mulai dari cell A1 s/d A3

SUMIF

=SUMIF(range, criteria, sum-range)
=SUMIF(A1:A8, "keju", B1:B8)
menjumlahkan cell dari B1 s/d B8 yang memenuhi kriteria "keju" dari cell A1 s/d A8

AVERAGE

=AVERAGE(A1-A3)
menghitung nilai rata2 dari mulai dari cell A1 s/d A3

COUNT

=COUNT(A1-A3)
menmenghitung banyaknya data pada cell A1 s/d A3

COUNTIF

=COUNTIF(A1-A3, 20)
menmenghitung berapa kali angka 20 muncul dari cell A1 s/d A3
=COUNTIF(A1-A3, "APEL")
menmenghitung berapa kali kata APEL muncul dari cell A1 s/d A3
(untuk mencari teks harus menggunakan kutip)
=COUNTIF(A1-A3, ">60")
menmenghitung berapa kali angka yang lebih besar dari 60 muncul dari cell A1 s/d A3
(jika kriteria menggunakan operator matematik harus menggunakan kutip)
